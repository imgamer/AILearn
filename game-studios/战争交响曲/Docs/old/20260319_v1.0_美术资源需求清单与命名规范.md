### 8.5 正则表达式

**文件名验证正则：**
```regex
^[a-z]{2,3}_[a-z0-9_]+_[a-z0-9]+_[a-z0-9]+_[a-z]{3,4}_v[0-9]{2}\.[a-z]{3}$
```

**解释：**
- `^[a-z]{2,3}_` - 前缀2-3个小写字母
- `[a-z0-9_]+` - 主体名称（小写字母、数字、下划线）
- `_[a-z0-9]+` - 部位/子对象
- `_[a-z0-9]+` - 用途后缀
- `_[a-z]{3,4}` - 通道标识（3-4个小写字母）
- `_v[0-9]{2}` - 版本号（v01-v99）
- `\.[a-z]{3}$` - 扩展名（3个小写字母）

### 8.6 禁止使用的特殊字符

| 字符 | 原因 |
|------|------|
| 空格 | 文件名不能有空格 |
| 中文 | 避免编码问题 |
| 特殊符号 `!@#$%^&*()+={}[]|\:;"'<>,?/` | 可能导致引擎或工具报错 |
| 连续下划线 `__` | 避免混淆 |
| 开头或结尾下划线 `_` | 避免混淆 |

**允许的字符：**
- 小写字母 a-z
- 数字 0-9
- 下划线 _

### 8.7 大小写策略

- **全部使用小写字母**
- 避免大小写敏感问题
- Windows/Linux/macOS统一处理

### 8.8 最大长度限制

- **文件名最大长度：64字符**
- 包含扩展名
- 超出会被自动截断或拒绝

### 8.9 版本号规则

| 版本号 | 说明 |
|--------|------|
| v01 | 初始版本 |
| v02-v09 | 小修改 |
| v10-v99 | 大修改 |

## 10. Python批处理脚本

### 10.1 资产校验脚本

```python
import os
import re
import csv
from pathlib import Path

class AssetValidator:
    def __init__(self):
        self.valid_files = []
        self.invalid_files = []
        self.pattern = re.compile(r'^[a-z]{2,3}_[a-z0-9_]+_[a-z0-9]+_[a-z0-9]+_[a-z]{3,4}_v[0-9]{2}\.[a-z]{3}$')
        self.max_length = 64
        self.allowed_extensions = ['tga', 'png', 'jpg', 'psd', 'ogg', 'wav', 'mp3']
        self.forbidden_chars = set(' !@#$%^&*()+={}[]|\\:;"\'<>,?/')
        
    def validate_filename(self, filename):
        if len(filename) > self.max_length:
            return False, f"文件名超过{self.max_length}字符限制"
        
        if any(char in filename for char in self.forbidden_chars):
            return False, "包含禁止字符"
        
        if not self.pattern.match(filename):
            return False, "不符合命名规范"
        
        ext = filename.split('.')[-1]
        if ext not in self.allowed_extensions:
            return False, f"不支持的扩展名: {ext}"
        
        return True, "有效"
        
    def scan_directory(self, root_path):
        root = Path(root_path)
        for file_path in root.rglob('*'):
            if file_path.is_file():
                is_valid, message = self.validate_filename(file_path.name)
                if is_valid:
                    self.valid_files.append({
                        'path': str(file_path),
                        'filename': file_path.name,
                        'status': 'valid',
                        'message': message
                    })
                else:
                    self.invalid_files.append({
                        'path': str(file_path),
                        'filename': file_path.name,
                        'status': 'invalid',
                        'message': message
                    })
    
    def generate_report(self, output_csv):
        with open(output_csv, 'w', newline='', encoding='utf-8') as f:
            writer = csv.DictWriter(f, fieldnames=['path', 'filename', 'status', 'message'])
            writer.writeheader()
            writer.writerows(self.valid_files + self.invalid_files)
        
        print(f"扫描完成:")
        print(f"  有效文件: {len(self.valid_files)}")
        print(f"  无效文件: {len(self.invalid_files)}")
        print(f"  报告已生成: {output_csv}")

if __name__ == '__main__':
    validator = AssetValidator()
    validator.scan_directory('Assets/Art')
    validator.generate_report('asset_validation_report.csv')
```

### 10.2 批量重命名脚本

```python
import os
import re
from pathlib import Path

class AssetRenamer:
    def __init__(self):
        self.renamed_files = []
        
    def parse_old_filename(self, filename):
        return {
            'prefix': 'chr',
            'subject': 'hero01',
            'part': 'body',
            'usage': 'd01',
            'channel': 'rgb',
            'version': 'v01',
            'ext': 'tga'
        }
    
    def generate_new_filename(self, parsed):
        return f"{parsed['prefix']}_{parsed['subject']}_{parsed['part']}_{parsed['usage']}_{parsed['channel']}_{parsed['version']}.{parsed['ext']}"
    
    def rename_file(self, old_path, new_name):
        old_path = Path(old_path)
        new_path = old_path.parent / new_name
        
        if new_path.exists():
            print(f"跳过（已存在）: {old_path.name} -> {new_name}")
            return False
        
        old_path.rename(new_path)
        self.renamed_files.append({
            'old': str(old_path),
            'new': str(new_path)
        })
        print(f"重命名: {old_path.name} -> {new_name}")
        return True
    
    def batch_rename(self, directory):
        root = Path(directory)
        for file_path in root.rglob('*'):
            if file_path.is_file():
                parsed = self.parse_old_filename(file_path.name)
                new_name = self.generate_new_filename(parsed)
                self.rename_file(file_path, new_name)

if __name__ == '__main__':
    renamer = AssetRenamer()
    renamer.batch_rename('Assets/Art')
```

### 10.3 资产清单生成脚本

```python
import os
import csv
from pathlib import Path
from collections import defaultdict

class AssetInventory:
    def __init__(self):
        self.inventory = defaultdict(list)
        
    def categorize_asset(self, filename):
        if filename.startswith('chr_'):
            return 'Characters'
        elif filename.startswith('env_'):
            return 'Environments'
        elif filename.startswith('item_'):
            return 'Items'
        elif filename.startswith('ui_'):
            return 'UI'
        elif filename.startswith('fx_'):
            return 'FX'
        elif filename.endswith('.ogg') or filename.endswith('.wav') or filename.endswith('.mp3'):
            return 'Audio'
        else:
            return 'Other'
    
    def scan_directory(self, root_path):
        root = Path(root_path)
        for file_path in root.rglob('*'):
            if file_path.is_file():
                category = self.categorize_asset(file_path.name)
                self.inventory[category].append({
                    'filename': file_path.name,
                    'path': str(file_path),
                    'size': file_path.stat().st_size,
                    'extension': file_path.suffix
                })
    
    def generate_inventory_csv(self, output_csv):
        with open(output_csv, 'w', newline='', encoding='utf-8') as f:
            writer = csv.writer(f)
            writer.writerow(['Category', 'Filename', 'Path', 'Size (bytes)', 'Extension'])
            
            for category in sorted(self.inventory.keys()):
                for asset in self.inventory[category]:
                    writer.writerow([
                        category,
                        asset['filename'],
                        asset['path'],
                        asset['size'],
                        asset['extension']
                    ])
        
        print(f"资产清单已生成: {output_csv}")
        for category, assets in self.inventory.items():
            print(f"  {category}: {len(assets)} 个文件")

if __name__ == '__main__':
    inventory = AssetInventory()
    inventory.scan_directory('Assets/Art')
    inventory.generate_inventory_csv('asset_inventory.csv')
```
